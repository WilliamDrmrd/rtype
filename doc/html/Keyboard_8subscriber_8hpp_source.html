39 class KeyboardReleasedSubscriber : public virtual ECS::EventSubscriber<KeyReleasedEvent>
40 {
41 public:
42 KeyboardReleasedSubscriber() = default;
43 ~KeyboardReleasedSubscriber() override = default;
44 void receiveEvent([[maybe_unused]] const std::string &name, const KeyReleasedEvent &data) override
45 {
46 std::string keyPressed = "";
47 Engine::System::InputsSystem *inputSystem =
48 dynamic_cast<Engine::System::InputsSystem *>(WORLD.getSystems()["InputsSystem"].get());
49 for (auto key : _keys)
50 if (data.keyEvent.code == key) {
51 if (inputSystem) {
52 if (data.keyEvent.code == sf::Keyboard::Period ||
53 data.keyEvent.code == sf::Keyboard::SemiColon)
54 keyPressed += ".";
55 else if (data.keyEvent.code >= sf::Keyboard::Num0 &&
56 data.keyEvent.code <= sf::Keyboard::Num9)
57 keyPressed += data.keyEvent.code - sf::Keyboard::Num0 + '0';
58 else if (data.keyEvent.code >= sf::Keyboard::Numpad0 &&
59 data.keyEvent.code <= sf::Keyboard::Numpad9)
60 keyPressed += data.keyEvent.code - sf::Keyboard::Numpad0 + '0';
61 else if (data.keyEvent.code == sf::Keyboard::Dash)
62 keyPressed += "6";
63 else if (data.keyEvent.code == sf::Keyboard::Quote)
64 keyPressed += "4";
65
66 if (data.keyEvent.code != sf::Keyboard::BackSpace)
67 inputSystem->changeText(keyPressed);
68 else
69 inputSystem->removeText();
70 }
71 }
72 }

76 sf::Keyboard::Num0, sf::Keyboard::Num1, sf::Keyboard::Num2, sf::Keyboard::Num3,
77 sf::Keyboard::Num4, sf::Keyboard::Num5, sf::Keyboard::Num6, sf::Keyboard::Num7,
78 sf::Keyboard::Num8, sf::Keyboard::Num9, sf::Keyboard::Numpad0, sf::Keyboard::Numpad1,
