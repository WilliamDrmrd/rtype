74 template <typename T> void registerComponent(ComponentType type)
75 {
76 components[type] = []() -> ECS::BaseComponent * { return new T(); };
77
78 destroyers[type] = [](ECS::Entity &entity) -> bool {
79 if (entity.has<T>()) {
80 entity.removeComponent<T>();
81 return true;
82 }
83 return false;
84 };
85
86 adders[type] = [](ECS::Entity &entity, ECS::BaseComponent *comp) -> bool {
87 if (entity.has<T>()) return false;
88 if (dynamic_cast<T *>(comp) == nullptr) return false;
89 entity.addComponent<T>(dynamic_cast<T *>(comp));
90 return true;
91 };
92 }

94 ECS::BaseComponent *createComponent(ComponentType type)
95 {
96 auto it = components.find(type);
97 if (it != components.end()) return it->second();
98 return nullptr;
99 }
