53 void setTexture()
54 {
55 if (!texture.loadFromFile(path)) throw std::runtime_error("Cannot load texture " + path);
56 sprite = sf::Sprite(texture);
57 sprite.setScale(scale.x, scale.y);
58 sprite.setPosition(position.x, position.y);
59 sprite.setRotation(rotation);
60 size = {texture.getSize().x, texture.getSize().y};
61 if (setOrigin) sprite.setOrigin(size.x / 2, size.y / 2);
62 }

64 RenderableComponent(const std::string &spriteName, int priority = 0, sf::Vector2<float> scale = {1, 1},
65 int rotation = 270)
66 : priority(priority), rotation(rotation), scale(scale), isDisplayed(true), name(spriteName)
67 {
68 auto it = spriteInfoMap.find(spriteName);
69 if (it == spriteInfoMap.end()) throw std::runtime_error("Cannot find sprite " + spriteName);
70 const SpriteInfo &info = it->second;
