318 template <typename T> id_t subscribe(const EventSubscriber<T> *subscriber)
319 {
320 auto type_id = ECS_TYPEID(T);
321 auto uuid = Utils::getNewId<T>();
322
323 if (_subscribers.find(type_id) == _subscribers.end())
324 _subscribers.emplace(type_id, std::unordered_map<id_t, BaseEventSubscriber *>());
325
326 _subscribers[type_id].emplace(
327 uuid, static_cast<BaseEventSubscriber *>(const_cast<EventSubscriber<T> *>(subscriber)));
328 return uuid;
329 }
330
337 template <typename T> void unsubscribe(id_t uuid)
338 {
339 auto type_id = ECS_TYPEID(T);
340
341 if (_subscribers.find(type_id) == _subscribers.end()) return;
342 _subscribers[type_id].erase(uuid);
343 }

415 void tick()
416 {
417 BaseSystem *renderer = nullptr;
418
419 for (int i = 0; i < Engine::EngineClass::getEngine().getPlayersAmount(); i++) {
420 Engine::EngineClass::getEngine().setCurrentPlayer(i);
421 for (auto &system : _systems) {
422 if (system.first == "Renderer") {
423 renderer = system.second.get();
424 continue;
425 }
426 if (system.second) system.second->tick();
427 }
428 }
429 Engine::EngineClass::getEngine().setCurrentPlayer(Engine::EngineClass::getEngine().getOwnPlayer());
430 if (renderer != nullptr) renderer->tick();
431 }
